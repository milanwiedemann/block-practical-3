Recode Variables

# RECODE ALL GAD VARIABLES AT THE SAME TIME ----
data_rec <- data %>% 
  mutate_at(
    c("gad_1", "gad_2", "gad_3", "gad_4", "gad_5", "gad_6", "gad_7"),
    funs(case_when(
      . == 1 ~ 0,
      . == 2 ~ 1,
      . == 3 ~ 2,
      . == 4 ~ 3)))
